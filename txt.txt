-- Quais são os produtos que já foram pedidos alguma vez ?
SELECT DISTINCT cd_produto, nm_produto
FROM tb_produto
WHERE cd_produto IN(SELECT cd_produto FROM tb_pedido);

-- Quais são os produtos nunca pedidos?
SELECT cd_produto, nm_produto
FROM tb_produto
WHERE cd_produto NOT IN(SELECT cd_produto FROM tb_pedido);
